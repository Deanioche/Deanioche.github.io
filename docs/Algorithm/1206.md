[BOJ 1260](https://www.acmicpc.net/problem/1260){: .btn .btn-blue .fs-2}

S2 \| 그래프 이론 \| 그래프 탐색 \| 너비 우선 탐색 \| 깊이 우선 탐색
{: .fs-2 .text-right}


```c++
#include <iostream>
#include <queue>
using namespace std;

int N, M, V;
int map[1001][1001];
bool visited[1001];

void DFS(int V)
{
    cout << V << ' ';
    visited[V] = true;

    for (int i = 1; i <= N; i++)
        if (map[V][i] && !visited[i])
            DFS(i);
}

void BFS(int V)
{
    queue<int> q;
    q.push(V);

    while (!q.empty())
    {
        V = q.front();
        q.pop();
        visited[V] = true;
        cout << V << ' ';
        for (int i = 1; i <= N; i++)
            if (map[V][i] && !visited[i])
            {
                q.push(i);
                visited[i] = true;
            }
    }
}

int main()
{
    cin.tie(0)->sync_with_stdio(0);
    cout.tie(0);

    cin >> N >> M >> V;
    for (int i = 0; i < M; i++)
    {
        int a, b;
        cin >> a >> b;
        map[a][b] = 1;
        map[b][a] = 1;
    }

    for (int i = 1; i <= N; i++)
        visited[i] = 0;
    DFS(V);
    cout << '\n';
    for (int i = 1; i <= N; i++)
        visited[i] = 0;
    BFS(V);
    return (0);
}
```
